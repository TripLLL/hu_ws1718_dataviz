Altersstruktur WP18 I

## Data preparation:
# Rough deciles
age.df <- mdb18.df %>%
  group_by(party) %>%
  mutate(rank = ntile(x = age, n = 10)) 

# Deciles 2-8
mid.age <- age.df %>%
  filter(rank > 1 & rank < 10)

# Decile 1 and 10
dev.age <- age.df %>%
  filter(rank == 1 | rank == 10)

## Plot:
plot.age <- ggplot(data = mid.age, aes(x = party, y = age)) +
  geom_jitter(stat = "identity", width = 0.1, color = "gray50", alpha = 0.2)
